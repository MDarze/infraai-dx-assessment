# UI Rework Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rebuild the InfraAI field assessment app's visual + interaction layer to match infraai-dx.com brand (cream/charcoal/teal, 2px geometry, IBM Plex Sans Arabic), Arabic-primary with locale toggle, system-theme toggle, desktop-first.

**Architecture:** Additive foundation first (new tokens/fonts/primitives/i18n landed alongside existing code so intermediate builds pass), then page reworks one at a time (Start → Survey → Results), then a final cleanup pass that tightens the Tailwind config and removes legacy bilingual strings. No component library; minimal i18n helper; self-hosted fonts.

**Tech Stack:** React 18, TypeScript, Vite 5, Tailwind CSS, Vitest + Testing Library (new), @fontsource packages (new).

**Spec:** `docs/superpowers/specs/2026-04-25-ui-rework-design.md` — source of truth for palette, typography, layouts, acceptance criteria.

**Repo:** `infraai-dx-assessment/` deploys `main` to Railway service `infraai-dx-dashboard`.

```ts
export type Theme = "system" | "light" | "dark";
export type ResolvedTheme = "light" | "dark";

const KEY = "infraai_theme";

export function getTheme(): Theme {
  const v = localStorage.getItem(KEY);
  return v === "light" || v === "dark" ? v : "system";
}

function systemPref(): ResolvedTheme {
  return window.matchMedia("(prefers-color-scheme: dark)").matches ? "dark" : "light";
}

export function getResolvedTheme(): ResolvedTheme {
  const t = getTheme();
  return t === "system" ? systemPref() : t;
}

export function setTheme(t: Theme): void {
  if (t === "system") localStorage.removeItem(KEY);
  else localStorage.setItem(KEY, t);
  document.documentElement.setAttribute("data-theme", t === "system" ? systemPref() : t);
}

export function cycleTheme(): Theme {
  const cur = getTheme();
  const next: Theme = cur === "system" ? "light" : cur === "light" ? "dark" : "system";
  setTheme(next);
  return next;
}
```

```ts
import { ar } from "./ar";
import { en } from "./en";

export type Locale = "ar" | "en";
const dicts = { ar, en } as const;
export type Key = keyof typeof ar;

const KEY = "infraai_locale";
let current: Locale = readInitial();

function readInitial(): Locale {
  const v = typeof localStorage !== "undefined" ? localStorage.getItem(KEY) : null;
  return v === "en" ? "en" : "ar";
}

export function getLocale(): Locale {
  return current;
}

export function setLocale(loc: Locale): void {
  current = loc;
  try {
    localStorage.setItem(KEY, loc);
  } catch {}
  if (typeof document !== "undefined") {
    document.documentElement.setAttribute("lang", loc);
    document.documentElement.setAttribute("dir", loc === "en" ? "ltr" : "rtl");
  }
  listeners.forEach((l) => l(loc));
}

type Listener = (loc: Locale) => void;
const listeners = new Set<Listener>();
export function subscribeLocale(l: Listener): () => void {
  listeners.add(l);
  return () => listeners.delete(l);
}

export function t(key: Key, vars?: Record<string, string | number>): string {
  const dict = dicts[current] as Record<string, string>;
  let s = dict[key] ?? key;
  if (vars) {
    for (const [k, v] of Object.entries(vars)) {
      s = s.replace(new RegExp(`\\{${k}\\}`, "g"), String(v));
    }
  }
  return s;
}
```

```tsx
import { useEffect, useState } from "react";
import { Card } from "../ui/Card";
import { Button } from "../ui/Button";
import { Radio } from "../ui/Radio";
import { Segmented } from "../ui/Segmented";
import { Field } from "../ui/Field";
import { t } from "../i18n/t";
import { useLocale } from "../i18n/useLocale";
import type { AssessmentState } from "../types";
import questionsRaw from "../questions.json";
import type { QuestionSet } from "../questions.types";

const questions = (questionsRaw as QuestionSet).questions;

interface Props {
  state: AssessmentState;
  activeRespondentId: string;
  setAnswer: (respondentId: string, questionId: string, value: unknown) => void;
  setNote: (respondentId: string, questionId: string, note: string) => void;
  onFinish: () => void;
}

export function Survey({ state, activeRespondentId, setAnswer, setNote, onFinish }: Props) {
  const [loc] = useLocale();
  const [idx, setIdx] = useState(0);
  const q = questions[idx];
  const respondent = state.respondents.find((r) => r.id === activeRespondentId);
  const currentValue = respondent?.answers[q.id] as string | undefined;
  const currentNote = (respondent?.notes?.[q.id] as string | undefined) ?? "";

  useEffect(() => {
    const onKey = (e: KeyboardEvent) => {
      if (document.activeElement instanceof HTMLInputElement) return;
      if (e.key === "Enter") next();
      else if (e.key === "Escape" && currentValue) setAnswer(activeRespondentId, q.id, "");
      else if (/^[1-9]$/.test(e.key) && q.options) {
        const i = parseInt(e.key, 10) - 1;
        const opt = q.options[i];
        if (opt) setAnswer(activeRespondentId, q.id, opt.value);
      }
    };
    window.addEventListener("keydown", onKey);
    return () => window.removeEventListener("keydown", onKey);
  });

  const next = () => {
    if (idx < questions.length - 1) setIdx(idx + 1);
    else onFinish();
  };
  const prev = () => setIdx(Math.max(0, idx - 1));

  const options = (q.options ?? []).map((o) => ({
    value: o.value,
    label: o[loc],
  }));

  return (
    <div className="mx-auto max-w-[720px] space-y-6">
      <div className="flex items-baseline justify-between text-xs text-ink-subtle">
        <span>{q.section[loc]}</span>
        <span>{t("survey.meta.count", { n: idx + 1, total: questions.length })}</span>
      </div>

      <Card>
        <h2 className="mb-6 text-xl font-medium text-ink">{q[loc].text}</h2>

        {q.kind === "single" && (
          <Radio
            name={q.id}
            value={currentValue}
            options={options}
            onChange={(v) => setAnswer(activeRespondentId, q.id, v)}
          />
        )}

        {q.kind === "scale" && (
          <Segmented
            value={currentValue}
            options={options}
            onChange={(v) => setAnswer(activeRespondentId, q.id, v)}
            ariaLabel={q[loc].text}
          />
        )}

        {q.kind === "text" && (
          <Field
            label=""
            value={currentValue ?? ""}
            onChange={(e) => setAnswer(activeRespondentId, q.id, e.target.value)}
          />
        )}

        <div className="mt-6">
          <Field
            label=""
            placeholder={t("survey.note.placeholder")}
            value={currentNote}
            onChange={(e) => setNote(activeRespondentId, q.id, e.target.value)}
          />
        </div>
      </Card>

      <div className="flex justify-between">
        <Button variant="secondary" onClick={prev} disabled={idx === 0}>
          {t("survey.previous")}
        </Button>
        <Button onClick={next}>
          {idx === questions.length - 1 ? t("survey.finish") : t("survey.next")}
        </Button>
      </div>
    </div>
  );
}
```
